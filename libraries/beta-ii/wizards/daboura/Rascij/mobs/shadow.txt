#include <std.h>
inherit MONSTER;
void create() {
   ::create();
	set_name("Shadow");
	set_id( ({ "shadow", "humanoid" }) );
	set_short("Shadow");
	set_long("A shadow of the forest");
	set_level(9);
	set_body_type("human");
	set_alignment(-5);
	set_race("demon");
	set_gender("male");
	set("aggressive", 2);
	set_wielding_limbs( ({ "left hand" }) );
	set_heart_beat(2);
	set_invis();
	set_moving(1);
	set_speed(3);
}
